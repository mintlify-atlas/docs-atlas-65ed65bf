<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-3xl font-bold text-gray-900 dark:text-white mb-3">Quick Start</h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-8">Get up and running with Kanban in minutes</p>
  
  <Steps>
    <Step title="Install Kanban">
      Install using your preferred method:
      
      <CodeGroup>
      ```bash Cargo
      cargo install kanban-cli
      ```
      
      ```bash Nix
      nix run github:fulsomenko/kanban
      ```
      
      ```bash Source
      git clone https://github.com/fulsomenko/kanban
      cd kanban
      cargo install --path crates/kanban-cli
      ```
      </CodeGroup>
    </Step>
    
    <Step title="Launch the TUI">
      Start the interactive terminal interface:
      
      ```bash
      kanban
      ```
      
      Or load a specific board file:
      
      ```bash
      kanban myboard.json
      ```
    </Step>
    
    <Step title="Create your first board">
      Press `n` to create a new board, then press `Enter` to activate it. Add cards with `n` and organize them across columns using vim-like navigation (`hjkl`).
      
      <Tip>Press `?` at any time to view all available keyboard shortcuts for your current context.</Tip>
    </Step>
  </Steps>
</div>
